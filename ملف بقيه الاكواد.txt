```csharp
// BlockFactory.Desktop/Infrastructure/GlobalExceptionHandler.cs

using System.IO;
using System.Windows;
using System.Windows.Threading;

namespace BlockFactory.Desktop.Infrastructure
{
    public static class GlobalExceptionHandler
    {
        private static readonly string LogFolder =
            Path.Combine(
                Environment.GetFolderPath(
                    Environment.SpecialFolder.LocalApplicationData),
                "BlockFactory", "Logs");

        public static void Register()
        {
            // ─── أخطاء UI Thread ─────────────────────
            Application.Current.DispatcherUnhandledException +=
                OnDispatcherUnhandledException;

            // ─── أخطاء Background Threads ────────────
            AppDomain.CurrentDomain.UnhandledException +=
                OnUnhandledException;

            // ─── أخطاء Async Tasks ───────────────────
            TaskScheduler.UnobservedTaskException +=
                OnUnobservedTaskException;

            Directory.CreateDirectory(LogFolder);
        }

        private static void OnDispatcherUnhandledException(
            object sender,
            DispatcherUnhandledExceptionEventArgs e)
        {
            e.Handled = true;
            LogError(e.Exception);
            ShowErrorDialog(e.Exception, false);
        }

        private static void OnUnhandledException(
            object sender,
            UnhandledExceptionEventArgs e)
        {
            var ex = e.ExceptionObject as Exception;
            if (ex == null) return;

            LogError(ex);

            if (e.IsTerminating)
                ShowErrorDialog(ex, true);
        }

        private static void OnUnobservedTaskException(
            object? sender,
            UnobservedTaskExceptionEventArgs e)
        {
            e.SetObserved();
            LogError(e.Exception);
        }

        // ─── تسجيل الخطأ ────────────────────────────
        private static void LogError(Exception ex)
        {
            try
            {
                var logFile = Path.Combine(
                    LogFolder,
                    $"error_{DateTime.Today:yyyyMMdd}.log");

                var logEntry =
                    $"\n{'═'.ToString().PadRight(60, '═')}\n" +
                    $"التاريخ: {DateTime.Now:dd/MM/yyyy HH:mm:ss}\n" +
                    $"النوع: {ex.GetType().Name}\n" +
                    $"الرسالة: {ex.Message}\n" +
                    $"المصدر: {ex.Source}\n" +
                    $"تتبع المكدس:\n{ex.StackTrace}\n";

                if (ex.InnerException != null)
                    logEntry +=
                        $"الخطأ الداخلي: " +
                        $"{ex.InnerException.Message}\n";

                File.AppendAllText(logFile, logEntry);
            }
            catch { }
        }

        // ─── عرض رسالة الخطأ ─────────────────────────
        private static void ShowErrorDialog(
            Exception ex, bool isFatal)
        {
            Application.Current.Dispatcher.Invoke(() =>
            {
                var message = isFatal
                    ? $"⚠️ خطأ فادح — سيتم إغلاق التطبيق\n\n" +
                      $"{GetFriendlyMessage(ex)}"
                    : $"⚠️ حدث خطأ غير متوقع\n\n" +
                      $"{GetFriendlyMessage(ex)}\n\n" +
                      $"تم حفظ تفاصيل الخطأ في ملف السجل.";

                var result = MessageBox.Show(
                    message,
                    isFatal ? "خطأ فادح" : "خطأ",
                    isFatal
                        ? MessageBoxButton.OK
                        : MessageBoxButton.OKCancel,
                    MessageBoxImage.Error,
                    MessageBoxResult.OK,
                    MessageBoxOptions.RightAlign |
                    MessageBoxOptions.RtlReading);

                if (isFatal)
                    Application.Current.Shutdown(1);

                if (result == MessageBoxResult.Cancel)
                    OpenLogFolder();
            });
        }

        private static string GetFriendlyMessage(Exception ex)
        {
            // تحويل رسائل الأخطاء التقنية لرسائل مفهومة
            if (ex.Message.Contains("SQL") ||
                ex.Message.Contains("database") ||
                ex.Message.Contains("connection"))
                return "خطأ في الاتصال بقاعدة البيانات.\n" +
                       "تأكد من أن النظام يعمل بشكل صحيح.";

            if (ex.Message.Contains("network") ||
                ex.Message.Contains("internet"))
                return "خطأ في الاتصال بالإنترنت.\n" +
                       "تحقق من اتصالك وأعد المحاولة.";

            if (ex.Message.Contains("access") ||
                ex.Message.Contains("permission"))
                return "خطأ في الصلاحيات.\n" +
                       "لا تملك صلاحية لتنفيذ هذا الإجراء.";

            return ex.Message.Length > 200
                ? ex.Message[..200] + "..."
                : ex.Message;
        }

        private static void OpenLogFolder()
        {
            try
            {
                System.Diagnostics.Process.Start(
                    "explorer.exe", LogFolder);
            }
            catch { }
        }
    }
}
```

## 📁 Part 29-C — Notification Service

```csharp
// BlockFactory.Desktop/Services/NotificationService.cs

using System.Collections.ObjectModel;
using System.Windows;
using System.Windows.Threading;

namespace BlockFactory.Desktop.Services
{
    public class NotificationService
    {
        private static NotificationService? _instance;
        public static NotificationService Instance =>
            _instance ??= new NotificationService();

        public ObservableCollection<AppNotification> Notifications { get; }
            = new();

        public event Action<AppNotification>? OnNotification;

        private NotificationService() { }

        // ─── إشعار نجاح ─────────────────────────────
        public void Success(string message, string? title = null)
        {
            Show(new AppNotification
            {
                Title = title ?? "تم بنجاح",
                Message = message,
                Type = NotificationType.Success,
                Icon = "✅",
                BackgroundColor = "#EAFAF1",
                ForegroundColor = "#27AE60"
            });
        }

        // ─── إشعار خطأ ──────────────────────────────
        public void Error(string message, string? title = null)
        {
            Show(new AppNotification
            {
                Title = title ?? "خطأ",
                Message = message,
                Type = NotificationType.Error,
                Icon = "❌",
                BackgroundColor = "#FDEDEC",
                ForegroundColor = "#E74C3C"
            });
        }

        // ─── إشعار تحذير ────────────────────────────
        public void Warning(string message, string? title = null)
        {
            Show(new AppNotification
            {
                Title = title ?? "تنبيه",
                Message = message,
                Type = NotificationType.Warning,
                Icon = "⚠️",
                BackgroundColor = "#FEF9E7",
                ForegroundColor = "#F39C12"
            });
        }

        // ─── إشعار معلومات ──────────────────────────
        public void Info(string message, string? title = null)
        {
            Show(new AppNotification
            {
                Title = title ?? "معلومة",
                Message = message,
                Type = NotificationType.Info,
                Icon = "ℹ️",
                BackgroundColor = "#EBF5FB",
                ForegroundColor = "#2980B9"
            });
        }

        private void Show(AppNotification notification)
        {
            Application.Current.Dispatcher.Invoke(() =>
            {
                Notifications.Insert(0, notification);
                OnNotification?.Invoke(notification);

                // حذف الإشعار بعد 4 ثواني
                var timer = new DispatcherTimer
                {
                    Interval = TimeSpan.FromSeconds(4)
                };
                timer.Tick += (s, e) =>
                {
                    timer.Stop();
                    Notifications.Remove(notification);
                };
                timer.Start();
            });
        }
    }

    public class AppNotification
    {
        public string Title { get; set; } = string.Empty;
        public string Message { get; set; } = string.Empty;
        public NotificationType Type { get; set; }
        public string Icon { get; set; } = string.Empty;
        public string BackgroundColor { get; set; } = "#FFFFFF";
        public string ForegroundColor { get; set; } = "#2C3E50";
        public DateTime CreatedAt { get; set; } = DateTime.Now;
    }

    public enum NotificationType
    {
        Success,
        Error,
        Warning,
        Info
    }
}
```

## 📁 Part 29-F — Additional Converters

```csharp
// BlockFactory.Desktop/Converters/AdditionalConverters.cs

using System.Globalization;
using System.Windows;
using System.Windows.Data;
using System.Windows.Media;

namespace BlockFactory.Desktop.Converters
{
    // ─── Initials Converter ──────────────────────────
    public class InitialsConverter : IValueConverter
    {
        public object Convert(object value, Type targetType,
            object parameter, CultureInfo culture)
        {
            var name = value?.ToString() ?? "";
            if (string.IsNullOrEmpty(name)) return "?";

            var parts = name.Split(' ',
                StringSplitOptions.RemoveEmptyEntries);

            return parts.Length >= 2
                ? $"{parts[0][0]}{parts[1][0]}"
                : name.Length >= 2
                    ? name[..2]
                    : name;
        }

        public object ConvertBack(object value, Type targetType,
            object parameter, CultureInfo culture)
            => throw new NotImplementedException();
    }

    // ─── Role Color Converter ────────────────────────
    public class RoleColorConverter : IValueConverter
    {
        public object Convert(object value, Type targetType,
            object parameter, CultureInfo culture)
        {
            return value?.ToString() switch
            {
                "Admin"      => "#1E3A5F",
                "Accountant" => "#2980B9",
                _ => "#95A5A6"
            };
        }

        public object ConvertBack(object value, Type targetType,
            object parameter, CultureInfo culture)
            => throw new NotImplementedException();
    }

    // ─── Role Name Converter ─────────────────────────
    public class RoleNameConverter : IValueConverter
    {
        public object Convert(object value, Type targetType,
            object parameter, CultureInfo culture)
        {
            return value?.ToString() switch
            {
                "Admin"      => "🔑 مدير",
                "Accountant" => "📊 محاسب",
                _ => value?.ToString() ?? "-"
            };
        }

        public object ConvertBack(object value, Type targetType,
            object parameter, CultureInfo culture)
            => throw new NotImplementedException();
    }

    // ─── Profit Color Converter ──────────────────────
    public class ProfitColorConverter : IValueConverter
    {
        public object Convert(object value, Type targetType,
            object parameter, CultureInfo culture)
        {
            if (value is decimal d)
                return d >= 0 ? "#EAFAF1" : "#FDEDEC";
            return "#F8F9FA";
        }

        public object ConvertBack(object value, Type targetType,
            object parameter, CultureInfo culture)
            => throw new NotImplementedException();
    }

    // ─── Null To Visibility ──────────────────────────
    public class NullToVisibilityConverter : IValueConverter
    {
        public object Convert(object value, Type targetType,
            object parameter, CultureInfo culture)
        {
            return value != null
                ? Visibility.Visible
                : Visibility.Collapsed;
        }

        public object ConvertBack(object value, Type targetType,
            object parameter, CultureInfo culture)
            => throw new NotImplementedException();
    }

    // ─── Bool To Text Converter ──────────────────────
    public class BoolToTextConverter : IValueConverter
    {
        public object Convert(object value, Type targetType,
            object parameter, CultureInfo culture)
        {
            bool boolValue = value is bool b && b;
            var parts = parameter?.ToString()?.Split('|')
                ?? new[] { "نعم", "لا" };

            return boolValue
                ? (parts.Length > 0 ? parts[0] : "نعم")
                : (parts.Length > 1 ? parts[1] : "لا");
        }

        public object ConvertBack(object value, Type targetType,
            object parameter, CultureInfo culture)
            => throw new NotImplementedException();
    }

    // ─── Bool To Color Converter ─────────────────────
    public class BoolToColorConverter : IValueConverter
    {
        public object Convert(object value, Type targetType,
            object parameter, CultureInfo culture)
        {
            bool boolValue = value is bool b && b;
            var parts = parameter?.ToString()?.Split('|')
                ?? new[] { "#27AE60", "#E74C3C" };

            var colorStr = boolValue
                ? (parts.Length > 0 ? parts[0] : "#27AE60")
                : (parts.Length > 1 ? parts[1] : "#E74C3C");

            return colorStr;
        }

        public object ConvertBack(object value, Type targetType,
            object parameter, CultureInfo culture)
            => throw new NotImplementedException();
    }

    // ─── Stock Color Converter ───────────────────────
    public class StockColorConverter : IValueConverter
    {
        public object Convert(object value, Type targetType,
            object parameter, CultureInfo culture)
        {
            if (value is int qty)
            {
                if (qty <= 0) return "#E74C3C";
                if (qty <= 100) return "#F39C12";
                return "#27AE60";
            }
            return "#95A5A6";
        }

        public object ConvertBack(object value, Type targetType,
            object parameter, CultureInfo culture)
            => throw new NotImplementedException();
    }

    // ─── Bool To Width Converter ─────────────────────
    public class BoolToWidthConverter : IValueConverter
    {
        public object Convert(object value, Type targetType,
            object parameter, CultureInfo culture)
        {
            bool boolValue = value is bool b && b;
            var parts = parameter?.ToString()?.Split('|')
                ?? new[] { "300", "0" };

            var widthStr = boolValue
                ? parts[0] : parts[1];

            return double.TryParse(widthStr, out double width)
                ? (object)new GridLength(width)
                : new GridLength(0);
        }

        public object ConvertBack(object value, Type targetType,
            object parameter, CultureInfo culture)
            => throw new NotImplementedException();
    }

    // ─── Inverse Bool To Visibility ──────────────────
    public class InverseBoolToVisibilityConverter : IValueConverter
    {
        public object Convert(object value, Type targetType,
            object parameter, CultureInfo culture)
        {
            bool boolValue = value is bool b && b;
            return boolValue
                ? Visibility.Collapsed
                : Visibility.Visible;
        }

        public object ConvertBack(object value, Type targetType,
            object parameter, CultureInfo culture)
            => throw new NotImplementedException();
    }

    // ─── Count To Bool ───────────────────────────────
    public class CountToBoolConverter : IValueConverter
    {
        public object Convert(object value, Type targetType,
            object parameter, CultureInfo culture)
        {
            if (value is int count) return count > 0;
            return false;
        }

        public object ConvertBack(object value, Type targetType,
            object parameter, CultureInfo culture)
            => throw new NotImplementedException();
    }

    // ─── Decimal To Bool ─────────────────────────────
    public class DecimalToBoolConverter : IValueConverter
    {
        public object Convert(object value, Type targetType,
            object parameter, CultureInfo culture)
        {
            if (value is decimal d) return d > 0;
            return false;
        }

        public object ConvertBack(object value, Type targetType,
            object parameter, CultureInfo culture)
        {
            if (value is Visibility v)
                return v == Visibility.Visible;
            return false;
        }
    }
}
```

```csharp
        public static IServiceProvider Services { get; private set; }
            = null!;

        protected override async void OnStartup(StartupEventArgs e)
        {
            base.OnStartup(e);

            // ─── تسجيل معالج الأخطاء العام ───────────
            GlobalExceptionHandler.Register();

            // ─── إعداد الـ Dependency Injection ──────
            Services = DependencyInjection.Configure();

            // ─── تهيئة قاعدة البيانات ─────────────────
            await InitializeDatabaseAsync();

            // ─── تشغيل النسخ الاحتياطي التلقائي ──────
            var backupService = GetService<BackupService>();
            backupService.Start();

            // ─── فتح نافذة تسجيل الدخول ───────────────
            var loginWindow = new LoginWindow();
            loginWindow.Show();
        }

        protected override void OnExit(ExitEventArgs e)
        {
            // إيقاف الـ API عند الإغلاق
            try
            {
                var apiService = GetService<ApiHostService>();
                if (apiService.IsRunning)
                    _ = apiService.StopAsync();
            }
            catch { }

            base.OnExit(e);
        }

        public static T GetService<T>() where T : notnull
            => Services.GetRequiredService<T>();

        private static async Task InitializeDatabaseAsync()
        {
            try
            {
                using var scope = Services.CreateScope();
                var context = scope.ServiceProvider
                    .GetRequiredService<AppDbContext>();

                // تطبيق الـ Migrations
                await context.Database.MigrateAsync();

                // تشغيل الـ Seeder
                await DatabaseSeeder.SeedAsync(context);
            }
            catch (Exception ex)
            {
                MessageBox.Show(
                    $"❌ خطأ في تهيئة قاعدة البيانات:\n\n{ex.Message}\n\n" +
                    "تأكد من تثبيت SQL Server LocalDB وأعد المحاولة.",
                    "خطأ في التشغيل",
                    MessageBoxButton.OK,
                    MessageBoxImage.Error,
                    MessageBoxResult.OK,
                    MessageBoxOptions.RightAlign |
                    MessageBoxOptions.RtlReading);

                Current.Shutdown(1);
            }
        }
    }
}
```

## 📁 Part 29-I — Auto-Save Service

```csharp
// BlockFactory.Desktop/Services/AutoSaveService.cs

using System.Windows.Threading;

namespace BlockFactory.Desktop.Services
{
    /// <summary>
    /// يحفظ أي تغييرات معلقة كل 30 ثانية
    /// حماية من انقطاع الكهرباء
    /// </summary>
    public class AutoSaveService
    {
        private readonly DispatcherTimer _timer;
        private readonly List<Func<Task>> _saveCallbacks = new();

        public AutoSaveService()
        {
            _timer = new DispatcherTimer
            {
                Interval = TimeSpan.FromSeconds(30)
            };
            _timer.Tick += async (s, e) => await ExecuteSavesAsync();
        }

        public void Start() => _timer.Start();
        public void Stop() => _timer.Stop();

        public void Register(Func<Task> saveCallback)
            => _saveCallbacks.Add(saveCallback);

        public void Unregister(Func<Task> saveCallback)
            => _saveCallbacks.Remove(saveCallback);

        private async Task ExecuteSavesAsync()
        {
            foreach (var save in _saveCallbacks.ToList())
            {
                try { await save(); }
                catch { }
            }
        }
    }
}
```
